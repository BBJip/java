001package de.dentrassi.varlink.maven;
002
003import java.io.IOException;
004import java.nio.file.Path;
005import java.util.LinkedList;
006import java.util.List;
007
008import org.eclipse.emf.common.util.URI;
009import org.eclipse.emf.ecore.EObject;
010import org.eclipse.emf.ecore.resource.Resource;
011import org.eclipse.emf.ecore.resource.ResourceSet;
012import org.eclipse.emf.ecore.resource.impl.ResourceSetImpl;
013
014import de.dentrassi.varlink.idl.VarlinkIdlStandaloneSetup;
015import de.dentrassi.varlink.idl.varlinkIdl.Interface;
016import de.dentrassi.varlink.idl.varlinkIdl.VarlinkIdlPackage;
017
018public class Loader {
019
020    public List<Interface> interfaces = new LinkedList<>();
021
022    public List<Interface> getInterfaces() {
023        return this.interfaces;
024    }
025
026    public void loadFrom(final Path path) throws IOException {
027
028        new VarlinkIdlStandaloneSetup().createInjectorAndDoEMFRegistration();
029
030        final ResourceSet resourceSet = new ResourceSetImpl();
031
032        final URI uri = URI.createURI(path.toUri().toString());
033
034        // initialize package
035        VarlinkIdlPackage.eINSTANCE.eClass();
036
037        final Resource r = resourceSet.createResource(uri);
038        r.load(null);
039
040        final EObject c = r.getContents().get(0);
041        if (!(c instanceof Interface)) {
042            throw new IllegalStateException(String.format("File '%s' does not contain a Varlink interface", path));
043        }
044
045        this.interfaces.add((Interface) c);
046    }
047
048}
